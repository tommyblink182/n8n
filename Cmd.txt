# Crea il namespace (se non esiste):
kubectl create namespace n8n

# Applica la PVC:
kubectl apply -f n8n-pvc.yaml

# Applica il deployment:
kubectl apply -f n8n-deployment.yaml

# Applica il service:
kubectl apply -f n8n-service.yaml

# Accedi a n8n 
http://localhost:30078⁠

# Applica ingress
kubectl apply -f n8n-ingress.yaml

# Verifica ingress
kubectl get ingress -n n8n

# Verifica i pod
kubectl get pod -n n8n

# Cancella i pod
kubectl delete pod -n n8n -l app=n8n
